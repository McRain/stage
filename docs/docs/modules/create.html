Создание ​

В приложении используется замечательный функционал библиотеки Module Federation

Мы рекомендуем познакомиться с этой библиотекой поближе, на официальном сайте, чтобы лучше понимать порядок подготовки и интеграции модулей.

Начнем ​

Создайте свой проект модуля и добавьте в него Module Federation как описано здесь

Мы будем показывать создание модулей на примере с использованием Webpack.

Добавьте @reneos/app пакет, он нужен для доступа к существующим классам

js
npm install @reneos/app --save

package.json

json

{
  "version": "0.0.1",
  "type": "module",
  "main": "/src/index.js",
  "scripts": {
    "start": "webpack serve --config webpack.config.js",
    "build": "webpack build --config webpack.prod.js"
  },
  "dependencies": {
    "@reneos/app": "latest"
  },
  "devDependencies": {
    "@module-federation/enhanced": "latest",
    "copy-webpack-plugin": "latest",
    "file-loader": "latest",
    "url-loader": "latest",
    "webpack": "latest",
    "webpack-cli": "latest",
    "webpack-dev-server": "latest"
  }
}

Конфигурация webpack.config.js

Конфигурация webpack.prod.js

js

import path from 'path'
import { fileURLToPath } from 'url'
import  { ModuleFederationPlugin } from '@module-federation/enhanced';

const __filename = fileURLToPath(import.meta.url)
const __dirname = path.dirname(__filename)

export default {
	mode: 'production',
	entry: './src/main.js',
	output: {
		path: path.resolve(__dirname, 'dist'),
		filename: 'bundle.js',
		publicPath: '/mods/troika/',
		clean: true
	},
	resolve: {
		extensions: ['.js', '.json'],
		alias: {
			"@": path.join(__dirname, './src'),
		},
	},
	optimization: {
		minimize: true,
		runtimeChunk: false
	},
	module: {
		rules: [
			{
				test: /\.(png|svg|jpg|jpeg|gif)$/i,
				type: 'asset/resource',
				generator: {
					filename: 'assets/[hash][ext][query]'
				}
			},
			{
				test: /\.json$/,
				type: 'json'
			}
		]
	},
	plugins: [
		new ModuleFederationPlugin({
			name: 'mymodulename',
			filename: 'mf.js',
			remotes: {
				'mymodule': 'mymodule@/mymodule/mod.mf.js',
			},
			exposes: {
				'./mymodule': './src/index.mf.js'
			},
			shared: {
				"@reneos/app": {
					import: "@reneos/app",
					singleton: true,
					eager: true,
					shareScope: 'default',
					shareKey: "@reneos/app"
				}
			}
		})
	]
}
